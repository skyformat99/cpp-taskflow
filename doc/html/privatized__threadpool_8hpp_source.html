54 // Privatized queue of worker. The lock-free queue is inspired by
55 // http://www.1024cores.net/home/lock-free-algorithms/queues/bounded-mpmc-queue
56 template<typename T, size_t C>
57 class PrivatizedWorkQueue {
58 
59  static_assert(C >= 2 && (C & (C -1)) == 0, "capacity must be a power of 2");
60 
61  public:
62 
63  PrivatizedWorkQueue() {
64  for (size_t i = 0; i < C; i ++){
65  _buffer[i].sequence_.store(i, std::memory_order_relaxed);
66  }
67  _front.store(0, std::memory_order_relaxed);
68  _back.store(0, std::memory_order_relaxed);
69  }
70 
71  ~PrivatizedWorkQueue(){}
72 
73  bool enqueue(T& data){
74  size_t pos = _front.load(std::memory_order_relaxed);
75  cell_t* cell = &_buffer[pos & (C-1)];
76  size_t seq = cell->sequence_.load(std::memory_order_acquire);
77  intptr_t dif = (intptr_t)seq - (intptr_t)pos;
78  if (dif == 0 && _front.compare_exchange_strong(pos, pos + 1, std::memory_order_relaxed)){
79  _num_items.fetch_add(1, std::memory_order_relaxed);
80  cell->data = std::move(data);
81  cell->sequence_.store(pos + 1, std::memory_order_release);
82  return true;
83  }
84  return false;
85  }
86 
87  bool dequeue(std::optional<T>& data){
88  if(empty()) return false;
89  size_t pos = _back.load(std::memory_order_relaxed);
90  cell_t* cell = &_buffer[pos & (C-1)];
91  size_t seq = cell->sequence_.load(std::memory_order_acquire);
92  intptr_t dif = (intptr_t)seq - (intptr_t)(pos + 1);
93  if (dif == 0 && _back.compare_exchange_strong(pos, pos + 1, std::memory_order_relaxed)){
94  _num_items.fetch_sub(1, std::memory_order_relaxed);
95  data = std::move(cell->data);
96  cell->sequence_.store(pos + C, std::memory_order_release);
97  return true;
98  }
99  return false;
100  }
101 
102  bool empty() const {
103  return _front.load(std::memory_order_relaxed) ==
104  _back.load(std::memory_order_relaxed);
105  };
106 
107  size_t size() const {
108  return _num_items.load(std::memory_order_relaxed);
109  }
110 
111  private:
112 
113  struct cell_t {
114  std::atomic<size_t> sequence_;
115  T data;
116  };
117 
118  static size_t const cacheline_size = 64;
119 
120  alignas(cacheline_size) std::array<cell_t, C> _buffer;
121  alignas(cacheline_size) std::atomic<size_t> _front;
122  alignas(cacheline_size) std::atomic<size_t> _back;
123  alignas(cacheline_size) std::atomic<size_t> _num_items {0};
124 };

219 // Function: _steal
220 template <typename Closure>
221 bool PrivatizedThreadpool<Closure>::_steal(std::optional<Closure>& w, uint64_t& seed, const size_t id){
222  const auto queue_num = _workers.size();
223  auto victim = _workers[id].last_victim;
224  for(size_t i=0; i<queue_num; i++){
225  if(_workers[victim].queue.dequeue(w)){
226  _workers[id].last_victim = victim;
227  return true;
228  }
229  victim += 1;
230  if(victim >= queue_num){
231  victim -= queue_num;
232  }
233  }
234  return false;
235 }

298 // Function: _spawn
299 template <typename Closure>
300 void PrivatizedThreadpool<Closure>::_spawn(unsigned N) {
301 
302  assert(is_owner());
303 
304  // Lock to synchronize all workers before creating _worker_mapss
305  std::scoped_lock lock(_mutex);
306 
307  for(size_t i=0; i<N; ++i){
308  _threads.emplace_back([this, i=i] () -> void {
309 
310  std::optional<Closure> t;
311  Worker& w = (_workers[i]);
312  uint64_t seed = i+1;
313  std::unique_lock lock(_mutex, std::defer_lock);
314 
315  while(!w.exit){
316  if(_num_tasks == 0 || (!w.queue.dequeue(t) && !_steal(t, seed, i))) {
317  lock.lock();
318  if(!_tasks.empty()) {
319  t = std::move(_tasks.back());
320  _tasks.pop_back();
321  }
322  else{
323  while(!w.exit && _num_tasks.load(std::memory_order_relaxed) == 0){
324  w.cv.wait(lock);
325  }
326  }
327  lock.unlock();
328  } // End of pop_front
329 
330  while(t) {
331  _num_tasks.fetch_sub(1, std::memory_order_relaxed);
332  (*t)();
333  if(w.cache) {
334  t = std::move(w.cache);
335  w.cache = std::nullopt;
336  }
337  else {
338  t = std::nullopt;
339  }
340  }
341  } // End of while ------------------------------------------------------
342 
343  });
344 
345  _worker_maps.insert({_threads.back().get_id(), i});
346  } // End of For ---------------------------------------------------------------------------------
347 }

349 template <typename Closure>
350 template <typename... ArgsT>
351 void PrivatizedThreadpool<Closure>::emplace(ArgsT&&... args){
352 
353  //no worker thread available
354  if(num_workers() == 0){
355  Closure{std::forward<ArgsT>(args)...}();
356  return;
357  }
358 
359  _num_tasks.fetch_add(1, std::memory_order_relaxed);
360 
361  Closure t {std::forward<ArgsT>(args)...};
362 
363  // caller is not the owner
364  if(auto tid = std::this_thread::get_id(); tid != _owner){
365 
366  // the caller is the worker of the threadpool
367  if(auto itr = _worker_maps.find(tid); itr != _worker_maps.end()){
368  // Speculation
369  if(!_workers[itr->second].cache.has_value()){
370  _workers[itr->second].cache = std::move(t);
371  return ;
372  }
373 
374  auto other = _fast_modulo(_randomize(_seed), _workers.size());
375  if(_workers[other].queue.size() > _workers[itr->second].queue.size()){
376  if(!_workers[itr->second].queue.enqueue(t)){
377  std::scoped_lock lock(_mutex);
378  _tasks.push_back(std::move(t));
379  }
380  }
381  else{
382  if(!_workers[other].queue.enqueue(t)){
383  std::scoped_lock lock(_mutex);
384  _tasks.push_back(std::move(t));
385  }
386  _workers[other].cv.notify_one();
387  }
388  return ;
389  }
390  }
391 
392  auto id = _fast_modulo(_randomize(_seed), _workers.size());
393 
394  if(!_workers[id].queue.enqueue(t)){
395  std::scoped_lock lock(_mutex);
396  _tasks.push_back(std::move(t));
397  }
398  else{
399  // Lock to make sure the worker will be notified
400  std::scoped_lock lock(_mutex);
401  }
402  _workers[id].cv.notify_one();
403 }

407 template <typename Closure>
408 void PrivatizedThreadpool<Closure>::batch(std::vector<Closure>&& tasks){
409 
410  //no worker thread available
411  if(num_workers() == 0){
412  for(auto &t: tasks){
413  t();
414  }
415  return;
416  }
417 
418  _num_tasks.fetch_add(tasks.size(), std::memory_order_relaxed);
419 
420  bool notify_all = tasks.size() > 1;
421  {
422  std::scoped_lock lock(_mutex);
423  std::move(tasks.begin(), tasks.end(), std::back_inserter(_tasks));
424  }
425 
426  if(!notify_all) {
427  auto id = _fast_modulo(_randomize(_seed), _workers.size());
428  _workers[id].cv.notify_one();
429  }
430  else {
431  for(size_t i=0; i<_workers.size(); ++i) {
432  _workers[i].cv.notify_one();
433  }
434  }
435 }
