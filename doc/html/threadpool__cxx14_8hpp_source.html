42  template<class T>
43  struct invoke_impl {
44  template<class F, class... Args>
45  static auto call(F&& f, Args&&... args)
46  -> decltype(std::forward<F>(f)(std::forward<Args>(args)...));
47  };

49  template<class B, class MT>
50  struct invoke_impl<MT B::*> {
51  template<class T, class Td = typename std::decay<T>::type,
52  class = typename std::enable_if<std::is_base_of<B, Td>::value>::type
53  >
54  static auto get(T&& t)->T&&;
55 
56  template<class T, class Td = typename std::decay<T>::type,
57  class = typename std::enable_if<is_reference_wrapper<Td>::value>::type
58  >
59  static auto get(T&& t) -> decltype(t.get());
60 
61  template<class T, class Td = typename std::decay<T>::type,
62  class = typename std::enable_if<!std::is_base_of<B, Td>::value>::type,
63  class = typename std::enable_if<!is_reference_wrapper<Td>::value>::type
64  >
65  static auto get(T&& t) -> decltype(*std::forward<T>(t));
66 
67  template<class T, class... Args, class MT1,
68  class = typename std::enable_if<std::is_function<MT1>::value>::type
69  >
70  static auto call(MT1 B::*pmf, T&& t, Args&&... args)
71  -> decltype((invoke_impl::get(std::forward<T>(t)).*pmf)(std::forward<Args>(args)...));
72 
73  template<class T>
74  static auto call(MT B::*pmd, T&& t)
75  -> decltype(invoke_impl::get(std::forward<T>(t)).*pmd);
76  };

202 // Procedure: spawn
203 // The procedure spawns "n" threads monitoring the task queue and executing each task. After the
204 // task is finished, the thread reacts to the returned signal.
205 inline void Threadpool::spawn(unsigned N) {
206 
207  if(is_worker()) {
208  throw std::runtime_error("Worker thread cannot spawn threads");
209  }
210 
211  for(size_t i=0; i<N; ++i) {
212 
213  _threads.emplace_back([this] () -> void {
214 
215  { // Acquire lock
216  std::lock_guard<std::mutex> lock(_mutex);
217  _worker_ids.insert(std::this_thread::get_id());
218  }
219 
220  bool stop {false};
221 
222  while(!stop) {
223  decltype(_task_queue)::value_type task;
224 
225  { // Acquire lock. --------------------------------
226  std::unique_lock<std::mutex> lock(_mutex);
227  _worker_signal.wait(lock, [this] () { return _task_queue.size() != 0; });
228  task = std::move(_task_queue.front());
229  _task_queue.pop_front();
230  } // Release lock. --------------------------------
231 
232  // Execute the task and react to the returned signal.
233  switch(task()) {
234  case Signal::SHUTDOWN:
235  stop = true;
236  break;
237 
238  default:
239  break;
240  };
241 
242  } // End of worker loop.
243 
244  { // Acquire lock
245  std::lock_guard<std::mutex> lock(_mutex);
246  _worker_ids.erase(std::this_thread::get_id());
247  }
248 
249  });
250  }
251 }

253 // Function: silent_async
254 // Insert a task without giving future.
255 template <typename C>
256 auto Threadpool::silent_async(C&& c, Signal sig) {
257 
258  // No worker, do this right away.
259  if(num_workers() == 0) {
260  c();
261  }
262  // Dispatch this to a thread.
263  else {
264  {
265  std::lock_guard<std::mutex> lock(_mutex);
266  _task_queue.emplace_back(
267  [c=std::forward<C>(c), ret=sig] () mutable {
268  c();
269  return ret;
270  }
271  );
272  }
273  _worker_signal.notify_one();
274  }
275 }

277 // Function: async
278 // Insert a callable task and return a future representing the task.
279 // Version for tasks returning void.
280 template<typename C>
281 std::enable_if_t<
282  std::is_same<void, std::invoke_result_t<C>>::value,
283  std::future<std::invoke_result_t<C>>
284 > Threadpool::async(C&& c, Signal sig) {
285 
286  using R = std::invoke_result_t<C>;
287 
288  std::promise<R> p;
289  auto fu = p.get_future();
290 
291  // No worker, do this immediately.
292  if(_threads.empty()) {
293  c();
294  p.set_value();
295  }
296  // Schedule a thread to do this.
297  else {
298  {
299  std::lock_guard<std::mutex> lock(_mutex);
300  _task_queue.emplace_back(
301  [p = MoC<decltype(p)>(std::move(p)), c = std::forward<C>(c), ret = sig]() mutable {
302  c();
303  p.get().set_value();
304  return ret;
305  }
306  );
307  }
308  _worker_signal.notify_one();
309  }
310  return fu;
311 }

347 // Procedure: wait_for_all
348 // After this method returns, all previously-scheduled tasks in the pool
349 // will have been executed.
350 inline void Threadpool::wait_for_all() {
351 
352  if(is_worker()) {
353  throw std::runtime_error("Worker thread cannot wait for all");
354  }
355 
356  std::mutex barrier_mutex;
357  std::condition_variable barrier_cv;
358  auto threads_to_sync{_threads.size()};
359  std::vector<std::future<void>> futures;
360 
361  auto barrier_task = [&] {
362  std::unique_lock<std::mutex> lock(barrier_mutex);
363  if (--threads_to_sync == 0) {
364  barrier_cv.notify_all();
365  }
366  else {
367  barrier_cv.wait(lock, [&threads_to_sync] {
368  return threads_to_sync == 0;
369  });
370  }
371  };
372 
373  for(size_t i=0; i<_threads.size(); ++i) {
374  futures.emplace_back(async(barrier_task));
375  }
376 
377  // Wait for all threads to have finished synchronization
378  for (auto& fu : futures) {
379  fu.get();
380  }
381 }

383 // Procedure: shutdown
384 // Remove a given number of workers. Notice that only the master can call this procedure.
385 inline void Threadpool::shutdown() {
386 
387  if(is_worker()) {
388  throw std::runtime_error("Worker thread cannot shut down the thread pool");
389  }
390 
391  for(size_t i=0; i<_threads.size(); ++i) {
392  silent_async([](){}, Signal::SHUTDOWN);
393  }
394 
395  for(auto& t : _threads) {
396  t.join();
397  }
398 
399  _threads.clear();
400 }
